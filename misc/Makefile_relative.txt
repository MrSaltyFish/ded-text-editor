CC = gcc
SRC = main.c
OUT = te
BUILD_DIR = build
SDL3_INC = -Isrc/include/SDL3
SDL3_LIB = -Lsrc/lib/SDL3 -lSDL3  # Assuming libraries are in src/lib/SDL3
CFLAGS = -Wall -Wextra -std=c11 -pedantic -ggdb $(SDL3_INC)
LIBS = $(SDL3_LIB) -lm

$(BUILD_DIR)/$(OUT): $(SRC)
	@mkdir -p $(BUILD_DIR)  # Create the build directory if it doesn't exist
	$(CC) $(CFLAGS) -o $(BUILD_DIR)/$(OUT) $(SRC) $(LIBS)

all: $(BUILD_DIR)/$(OUT)

clean:
	rm -rf $(BUILD_DIR)
